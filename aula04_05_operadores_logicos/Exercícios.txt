		i. a variação dos anos (o X) vai do 1º até o 3º

    print(f"O número {num} é Negativo!")
else:
    print(f"O número {num} é Zero!")

# 2 Par ou Ímpar
if num % 2 == 0:
    print(f"O número {num} é Par!")
else:
    print(f"O número {num} é Ímpar!")

    # 3 Múltiplos de 3, 5 ou 7 ( Somente se for Ímpar)

    # Variável de controle
    multiplo_encontrado = False

    if num % 3 == 0:
        print(" É múltiplo de 3.")
        multiplo_encontrado = True

    if num % 5 == 0:
        print(" É múltiplo de 5.")
        multiplo_encontrado = True

    if num % 7 == 0:
        print(" É múltiplo de 7.")
        multiplo_encontrado = True

    if not multiplo_encontrado:
        print(" NÃO é múltiplo de 3, 5 ou 7.")

''' 4. Crie uma calculadora. Seu programa deve ter 3 inputs. O primeiro pegará um número float, o segundo irá pegar a operação aritmética a ser realizada (adição, subtração, multiplicação, divisão e potência) e o terceiro input deve pegar o segundo número em formato float. Mostre o resultado da operação em um texto completo, como abaixo:
a. A equação é : 2 ** 6 = 64
b. Onde o primeiro input foi 2, o segundo input foi ** e o terceiro foi 6.
c. Avise o usuário caso a operação inserida seja inválida.
d. Se a operação for de divisão e o segundo número for 0, avise “Impossível dividir por zero”.
'''

# 1. Primeiro número 
num1 = float(input('Digite o primeiro número: '))

# 2. Operação
operacao = input('Digite a operação (+, -, *, /, **): ')

# 3. Segundo  número
num2 = float(input('Digite o segundo número: '))

resultado = None
operacao_valida = True

# Lógica da Calculadora
if operacao == '+':
    resultado = num1 + num2

elif operacao == '-':
    resultado = num1 - num2
elif operacao == '*':
    resultado = num1 * num2
elif operacao == '/':
    # d. Tratar divisão por zero
    if num2 == 0:
        print('\nAVISO: Impossível dividir por zero')
        operacao_valida = False # Marcar como inválida para não mostrar a equação
    else:
        resultado = num1 / num2
elif operacao == '**':
    resultado = num1 ** num2
# c. Tratar operação inválida
else:
    print('\nAVISO: Operação inserida é inválida.')
    operacao_valida = False

# Mostrar o resultado
if operacao_valida:
    # a. e b. Mostrar o resultado no formato completo
    print(f'\nA equação é {num1} {operacao} {num2} = {resultado}')
    print(f'Onde o primeiro input foi {num1}, o segundo input foi {operacao} e o terceiro input foi {num2}.')

''' 5. Crie um algoritmo para determinar para onde alguém deve ir com base na sua idade. Abaixo está o critério que determina para onde ir:
	a. se idade for menor que 3 -> vá para o Berçário
	b. se idade for de 3 até 5 -> vá para a Educação Infantil na Série NX
		i. a variação das séries (o X) vai do N1 até o N3
	c. se idade for de 6 até 14 -> vá para o Ensino Fundamental no Ano X
		i. a variação dos anos (o X) vai do 1º até o 9º
	d. se idade for de 15 até 17 -> vá para o Ensino Médio no Ano X
		i. a variação dos anos (o X) vai do 1º até o 3º
	e. se idade for maior que 17 -> vá para a faculdade
	f. Exemplo: se o usuário digitar 13, a saída deverá ser “Vá para o Ensino Fundamental no Ano 8”
'''

# Assumimos que o usuário digita um número inteiro
idade = int(input('Digite a idade para determinar o destino: '))
destino = ""

# a. Berçário
if idade < 3:
    destino = "Berçário"

#  b. Educação Infantil ( 3 a 5 anos)
elif idade >= 3 and idade <= 5:
    # Série X = idade -2 ( N1. N2. N3)
    serie_x = idade - 2
    destino = f"Educação Infantil na Série {serie_x}"


# c. Ensino Fundamental ( 6 a 14 anos)
elif idade >= 6 and idade <= 14:
    # Ano X = idade - 5 (1º até 9ª)
    ano_x = idade -5
    destino = f"Ensino Fundamental no Ano {ano_x}º"


# d. Ensino Médio ( 15 a 17 anos)
elif idade >= 15 and idade <= 17:
    # Ano X = idade - 14 ( 1º até 3º)
    ano_x = idade - 14
    destino = f"Ensino Médio no Ano {ano_x}º"


# e. Faculdade ( maior que 17 anos)
else:
    destino = "Faculdade"


# Exibir o resultado final
print(f"Sua idade é de {idade}, assim você irá para: {destino}.")
